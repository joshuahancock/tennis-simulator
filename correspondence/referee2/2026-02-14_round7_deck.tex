\begin{frame}[t]{Verdict: Accept with Minor Revisions}

\textbf{The project is methodologically sound.} Data integrity is confirmed, diagnostics are honest, and 17 analysis scripts cover extensive ground.

\vspace{0.8em}
\textbf{Key findings this round:}

\begin{itemize}
  \item \textbf{K-factor tested}: K=64 gives +0.4pp accuracy but worse calibration. Trajectory lag not solved by K alone.
  \item \textbf{Calibration curve}: Systematic overconfidence on favorites, underconfidence on underdogs. This is the most exploitable signal.
  \item \textbf{``Agree but less confident''}: +6.4\% ROI (N=256, p=0.041) --- promising but needs multi-year validation.
  \item \textbf{Multiple testing}: 50+ subsets tested without correction. The p=0.041 signal is not distinguishable from noise in isolation.
\end{itemize}

\vspace{0.5em}
\textbf{Core recommendation:} Shift from broad scanning to \textbf{depth on one signal}. Implement Platt scaling, validate across years, paper-trade prospectively.

\end{frame}

\end{table}

\vspace{0.6em}
\textbf{Pattern:} Elo overestimates the gap between favorites and underdogs by 6+ percentage points at the extremes.

\column{0.48\textwidth}

\textbf{Why this matters for betting:}

\begin{itemize}
  \item If Elo says 80\%, reality is $\sim$72\%
  \item If Elo says 25\%, reality is $\sim$32\%
  \item Underdogs are \textit{worth more} than Elo thinks
  \item Favorites are \textit{worth less} than Elo thinks
\end{itemize}

\vspace{0.6em}
\textbf{Fix:} Platt scaling \\
\texttt{cal = plogis(a + b*qlogis(elo))} \\
\textit{5 lines of R code. Not yet implemented.}

\end{columns}
\end{frame}

\begin{frame}[t]{50+ Subset Tests Require Multiple Testing Correction}

\footnotesize
\begin{columns}[T]
\column{0.45\textwidth}

\textbf{Subsets tested (partial list):}
\begin{itemize}\setlength\itemsep{0pt}
  \item 4 K-factor values
  \item 4+ edge thresholds
  \item Agreement/disagreement
  \item 5+ confidence buckets
  \item 4 tournament levels, 3 surfaces
  \item 4+ odds ranges
  \item Handedness, height, fatigue
  \item Match load, Clay Masters
  \item Various combinations
\end{itemize}

\column{0.50\textwidth}

\textbf{At $\alpha$ = 0.05 with 50 tests:}
\begin{itemize}\setlength\itemsep{0pt}
  \item Expected false positives: $\sim$2.5
  \item Bonferroni threshold: 0.001
  \item Best p-value found: 0.041
  \item \textcolor{negred}{Does not survive correction}
\end{itemize}

\vspace{0.4em}
\textbf{This does not mean the signal is false.}

\vspace{0.2em}
It means single-period evidence is insufficient. The correct response is \textbf{multi-year replication}, not tighter p-values from the same data.

\end{columns}
\end{frame}
